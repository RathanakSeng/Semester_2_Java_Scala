class MainActivity : AppCompatActivity()
{
    override fun onCreate(savedInstanceState: Bundle?)
    {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

        Explicit.setOnClickListener()
        {
            var msg = message.text.toString()

            var intent = Intent()
            intent.action = Intent.ACTION_SEND
            intent.type = "text/plain" //What we want to share
            intent.putExtra(Intent.EXTRA_TEXT, msg) //This will show in the sharing content

            startActivity(Intent.createChooser(intent, "Share With Friend..."))
        }
    }
}
